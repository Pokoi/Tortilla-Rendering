61  // Code based in code snippet from tinyobjloader repository README
62  // https://github.com/syoyo/tinyobjloader/blob/master/README.md

65  tinyobj::attrib_t attrib;
66  std::vector<tinyobj::shape_t> shapes;
67  std::vector<tinyobj::material_t> materials;
68 
69  std::string warn;
70  std::string err;
71 
72  if (tinyobj::LoadObj(&attrib, &shapes, &materials, &warn, &err, mesh_path.c_str()))
73  {
74  // For each tinyobj shape
75  for (size_t s = 0; s < shapes.size(); ++s)
76  {
77  size_t index_offset = 0;
78 
79  std::vector<int> indices;
80 
81  indices.resize(shapes[s].mesh.indices.size());
82 
83  // Calculate the vertex and normal for each index
84  for (int index = 0; index < indices.size(); ++index)
85  {
86  int vertex_index = shapes[s].mesh.indices[index].vertex_index;
87  int normal_index = shapes[s].mesh.indices[index].normal_index;
88 
89  tinyobj::real_t vx = attrib.vertices[3 * vertex_index + 0];
90  tinyobj::real_t vy = attrib.vertices[3 * vertex_index + 1];
91  tinyobj::real_t vz = attrib.vertices[3 * vertex_index + 2];
92  tinyobj::real_t nx = attrib.normals[3 * normal_index + 0];
93  tinyobj::real_t ny = attrib.normals[3 * normal_index + 1];
94  tinyobj::real_t nz = attrib.normals[3 * normal_index + 2];
95 
96  original_vertices.push_back ( { {vx, vy, vz, 1} } );
97  original_normals.push_back ( { {nx, ny, nz, 1} } );
98 
99  indices[index] = index_offset + index;
100  }
101 
102  index_offset += indices.size();
103  std::shared_ptr<Mesh> mesh = std::make_shared<Mesh>(indices, this);
104  meshes.push_back(mesh);
105  }
106  }
107 
108  // Assign the size to the collections
109  transformed_vertices.resize(original_vertices.size());
110  transformed_normals.resize(original_normals.size());
111  display_vertices.resize(original_vertices.size());

130  void Model::Update(float delta, View & view)
131  {
132  transform->update_transform();
133 
134  // Apply camera transformations
135  transformation = view.get_camera().get_transform().get_inverse_transformation() * toolkit::Matrix44f(view.get_camera().get_projection()) * toolkit::Matrix44f(get_transform().get_transformation());
136 
137  // Transformation per vertex
138  for (size_t index = 0; index < original_vertices.size(); ++index)
139  {
140  toolkit::Point4f& vertex = transformed_vertices[index] = toolkit::Matrix44f(transformation) * toolkit::Matrix41f(original_vertices[index]);
141  toolkit::Vector4f& normal = transformed_normals[index] = toolkit::Matrix44f(transformation) * toolkit::Matrix41f(original_normals[index]);
142 
143  // Normalize last value for perspective transformation
144  float vertex_divisor = 1.f / vertex[3];
145 
146  vertex[0] *= vertex_divisor;
147  vertex[1] *= vertex_divisor;
148  vertex[2] *= vertex_divisor;
149  vertex[3] = 1.f;
150 
151  float normal_divisor = 1.f / normal[3];
152 
153  normal[0] *= normal_divisor;
154  normal[1] *= normal_divisor;
155  normal[2] *= normal_divisor;
156  normal[3] = 1.f;
157 
158  float v = std::sqrt(std::pow(normal[0], 2) + std::pow(normal[1], 2) + std::pow(normal[2], 2));
159  v = 1 / v;
160  normal[0] *= v;
161  normal[1] *= v;
162  normal[2] *= v;
163  normal[3] = 0.f;
164  }
165  }
