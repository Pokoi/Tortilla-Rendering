118  bool Mesh::is_frontface(const toolkit::Point4f* const projected_vertices, const int* const indices)
119  {
120  const toolkit::Point4f & v0 = projected_vertices[indices[0]];
121  const toolkit::Point4f & v1 = projected_vertices[indices[1]];
122  const toolkit::Point4f & v2 = projected_vertices[indices[2]];
123 
124  return ((v1[0] - v0[0]) * (v2[1] - v0[1]) - (v2[0] - v0[0]) * (v1[1] - v0[1]) > 0.f);
125  }

131  void Mesh::illuminate(View& view)
132  {
133  auto normals = model->get_original_normals();
134  auto & transformed_normals = model->get_transformed_normals();
135 
136  for (int i = 0; i < indices.size(); ++i)
137  {
138  Color_Buffer_Rgba8888::Color diffuse = material.get_color();
139  Color_Buffer_Rgba8888::Color ambient = view.get_ambient_color();
140 
141  float r_component = 0;
142  float g_component = 0;
143  float b_component = 0;
144 
145  diffuse * material.get_kd();
146 
147  // Directional lights illumination
148  for (std::shared_ptr<DirectionalLight> & light : view.get_directional_lights())
149  {
150  Color_Buffer_Rgba8888::Color light_color = light->get_light_color(model->get_transformed_vertices()[indices[i]]);
151  toolkit::Vector4f direction = light->get_direction(model->get_transformed_vertices()[indices[i]]);
152 
153  float multiplier = model->get_transformed_normals()[indices[i]].dot_product(direction);
154 
155  if (multiplier > 1)
156  {
157  multiplier = 1;
158  }
159 
160  else if (multiplier < 0)
161  {
162  multiplier = 0;
163  }
164 
165  light_color * material.get_kl();
166 
167  r_component += ((diffuse.data.component.r * light_color.data.component.r) >> 8) * multiplier;
168  g_component += ((diffuse.data.component.g * light_color.data.component.g) >> 8) * multiplier;
169  b_component += ((diffuse.data.component.b * light_color.data.component.b) >> 8) * multiplier;
170 
171 
172  }
173 
174  // Point Lights illumination
175  for (std::shared_ptr<PointLight>& light : view.get_point_lights())
176  {
177 
178  Color_Buffer_Rgba8888::Color light_color = light->get_light_color(model->get_transformed_vertices()[indices[i]]);
179  toolkit::Vector4f direction = light->get_direction(model->get_transformed_vertices()[indices[i]]);
180 
181  float multiplier = model->get_transformed_normals()[indices[i]].dot_product(direction);
182 
183  if (multiplier > 1)
184  {
185  multiplier = 1;
186  }
187  else if (multiplier < 0)
188  {
189  multiplier = 0;
190  }
191 
192  light_color* material.get_kl();
193 
194  r_component += ((diffuse.data.component.r * light_color.data.component.r) >> 8) * multiplier;
195  g_component += ((diffuse.data.component.g * light_color.data.component.g) >> 8) * multiplier;
196  b_component += ((diffuse.data.component.b * light_color.data.component.b) >> 8) * multiplier;
197  }
198 
199  // Add ambient component
200  r_component += ambient.data.component.r * material.get_ka();
201  g_component += ambient.data.component.g * material.get_ka();
202  b_component += ambient.data.component.b * material.get_ka();
203 
204  // Set the final color
205  material.get_transformed_colors()[indices[i]].set( r_component, g_component, b_component );

213  void Mesh::NDC_transformation()
214  {

216 
217  for (int i = 0; i < indices.size(); ++i)

233  void Mesh::display_coordinates_transformation(size_t width, size_t height)
234  {
235  toolkit::Scaling3f scaling = toolkit::Scaling3f(float(width / 2), float(height / 2), 100000000.f);
236  toolkit::Translation3f translation = toolkit::Translation3f(float(width / 2), float(height / 2), 0.f);
237 
238  toolkit::Transformation3f transformation = translation * scaling;
239 
240  for (int i = 0; i < indices.size(); ++i)
241  {
242  model->get_display_vertices()[indices[i]] = Point4i(toolkit::Matrix44f(transformation) * toolkit::Matrix41f(model->get_transformed_vertices()[indices[i]]));
243  }
